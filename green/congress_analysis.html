[1] tokenize

create tokens
##### TOKENS #####
enviroTokens_dtm <- enviroSpeeches %>%
  corpus(docid_field="speech_id",
         text_field="speech") %>%
  tokens(remove_punct=TRUE, #CREATE TOKEN OBJECT
         remove_symbols=TRUE,
         remove_numbers=TRUE,
         remove_separators=TRUE,
         verbose=TRUE) %>% 
  tokens_remove(stopwords, #REMOVE STOP WORDS
                verbose=TRUE) %>%  
  dfm() #CREATE DOC-TERM MATRIX
create bigrams
##### BIGRAMS #####
enviroBigrams_dtm <- enviroSpeeches %>%
  corpus(docid_field="speech_id",
         text_field="speech") %>%
  tokens(remove_punct=FALSE, #CREATE TOKEN OBJECT
         remove_symbols=TRUE,
         remove_numbers=TRUE,
         remove_separators=TRUE,
         padding=FALSE,
         verbose=TRUE) %>% 
  tokens_remove("\\p{P}", valuetype = "regex", #REMOVE PUNCT
                padding=TRUE, #REPLACE W/ EMPTY STRINGS
                verbose=TRUE) %>% 
  tokens_remove(stopwords, #REMOVE STOP WORDS BEFORE BIGRAMS CREATED
                padding=FALSE, 
                verbose=TRUE) %>%  
  tokens_ngrams(n=2) %>% #CREATE BIGRAMS
  dfm() #CREATE DOC-TERM MATRIX
create triples
##### TRIPLES #####
enviroTriples_dtm <- enviroSpeeches %>%

                padding=FALSE, 
                verbose=TRUE) %>%

  dfm() #CREATE DOC-TERM MATRIX

enviroTokens <- enviroTokens_dtm %>%
  tidy() %>%
  mutate(document=as.double(document)) %>%
  left_join(enviroSpeeches,by=join_by(document==speech_id)) %>%
  select(-speech) %>%
  select(year,date,document,word_count,chamber:gender,term,count)
